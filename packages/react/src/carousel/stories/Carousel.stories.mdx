# Carousel

Horizontal scrollable areas used for displaying grouped content.

### Installation

```bash
npm install --save @stewed/react
```

### Import

```js
import { Carousel } from '@stewed/react';
```

## Examples

export const Template = (args) => (
    <Theme>
        <Carousel {...args}>
            {Array.from({ length: 6 }).map((_, index) => (
                <div
                    key={index}
                    style={{
                        display: 'flex',
                        justifyContent: 'center',
                        alignItems: 'center',
                        fontSize: 60,
                        backgroundColor: '#ddd',
                        width: '100%',
                        height: '300px',
                    }}
                >
                    {index + 1}
                </div>
            ))}
        </Carousel>
    </Theme>
);
